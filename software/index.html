const mDKey = require('../models/metadataKey');

exports.getAllMDKeys = async (req, res) => {
  try {
    const mDKeys = await mDKey.findAll();
    res.status(200).json({
      status: 'success',
      data: { mDKeys },
    });
  } catch (e) {
    res.json({ message: 'Failed to get all metadata keys' });
  }
};
